const addXP = async () => {
  doHaptic();
  try {
    const userName = await AsyncStorage.getItem('user_name') || 'TestUser';

## 📝 ТЕХНИЧЕСКОЕ ОБЪЯСНЕНИЕ

### Что такое `addOrUpdateScore()`?

Это **главная функция**, которая обновляет ВСЕ структуры данных при добавлении XP:

```typescript
export const addOrUpdateScore = async (
  userName: string,
  pointsDelta: number,
  lang: 'ru' | 'uk' = 'ru'
): Promise<void> => {
  // 1. Обновляет leaderboard (кто в топе)
  // 2. Обновляет week_points_v2 (очки за неделю)
  // 3. Обновляет week_leaderboard (рейтинг недели)
  // 4. Обновляет daily_stats (график по дням)
  // 5. Обновляет стрик (активность)
};
```

Эта функция вызывается:
- Во всех нормальных источниках XP (уроки, квизы, экзамены)
- В системе активного повторения (review)
- В дневных задачах

Теперь она вызывается и в тестер-функции! ✅
